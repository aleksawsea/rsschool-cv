const addTagsClickHandler = () => {
    document.querySelector('.gifts-filter').addEventListener('pointerdown', (event) => {
    if (event.target.classList.contains('filter-item')) {
        let clickedTag = event.target;
        removeSelectedTags();
        selectClickedTag(clickedTag);
        filterCardsByCategory(clickedTag.innerText);
        }
    })
}

const removeSelectedTags = () => {
    let tags = document.querySelectorAll('.gifts-filter .filter-item');
    tags.forEach(tag => {
     tag.classList.remove('active');
    })
}

const selectClickedTag = (clickedTag) => {
    clickedTag.classList.add('active');
}

const filterCardsByCategory = (selectedTag) => {
    let gifts = document.querySelectorAll('.gifts-content .gift-card');
    gifts.forEach(gift => {
        gift.classList.add('gift-hidden');
        gift.querySelectorAll('.gift-card .gift-description h4').forEach(name => {
            if (name.innerText.toLowerCase() === selectedTag.toLowerCase()) {
                gift.classList.remove('gift-hidden');
            } else if (selectedTag === 'ALL') {
                gift.classList.remove('gift-hidden');
            }
        })
    })
}
